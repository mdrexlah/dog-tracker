        function showTab(tabName) {

            document.querySelectorAll('.tab-content').forEach(tab => { tab.classList.remove('active'); });

        function setupSheets() {

            const url = document.getElementById('sheet-url').value.trim();

            if (!url.includes('docs.google.com/spreadsheets')) {

                showStatus('setup-status', 'Please enter a valid Google Sheets URL', 'error');

                return;

            }

            const match = url.match(/\/spreadsheets\/d\/([a-zA-Z0-9-_]+)/);

            if (match) {

                SHEET_ID = match[1];

                SHEET_URL = url;

                localStorage.setItem('michaels_dogs_sheet_id', SHEET_ID);

                localStorage.setItem('michaels_dogs_sheet_url', SHEET_URL);

                document.getElementById('view-sheets-link').href = SHEET_URL;

                showStatus('setup-status', '✅ Connected to Google Sheets! You can now start entering data.', 'success');

                initializeSheets();

            } else {

                showStatus('setup-status', 'Could not extract Sheet ID from URL. Please check the URL.', 'error');

            }

        }

            showStatus('setup-status', '📋 Sheet structure ready! Start entering your business data.', 'success');

        }

        function addMileage() {

            const date = document.getElementById('mileage-date').value;

            const miles = document.getElementById('mileage-miles').value;

            const purpose = document.getElementById('mileage-purpose').value;

            if (!date || !miles) {

                showStatus('mileage-status', 'Please fill in date and miles', 'error');

                return;

            }

            const deduction = (parseFloat(miles) * 0.67).toFixed(2);

            saveToGoogleSheets('Mileage', [date, miles, purpose, deduction]);

            showStatus('mileage-status', `✅ Saved: ${miles} miles = $${deduction} deduction`, 'success');

            document.getElementById('mileage-miles').value = '';

            document.getElementById('mileage-purpose').value = '';

            updateWeeklySummary();

        }

        function saveToGoogleSheets(sheetName, data) {

            const key = `michaels_dogs_${sheetName.toLowerCase()}`;

            let existing = JSON.parse(localStorage.getItem(key) || '[]');

            existing.push(data);

            localStorage.setItem(key, JSON.stringify(existing));

            console.log(`Saving to Google Sheets - ${sheetName}:`, data);

        }

        function showStatus(elementId, message, type) {

            const element = document.getElementById(elementId);

            element.textContent = message;

            element.className = `status-message ${type}`;

            element.style.display = 'block';

            setTimeout(() => {

                element.style.display = 'none';

            }, 5000);

        }

        function updateWeeklySummary() {

            const mileageData = JSON.parse(localStorage.getItem('michaels_dogs_mileage') || '[]');

            const thisWeek = getThisWeekData(mileageData);

            const weeklyDeduction = thisWeek.reduce((sum, entry) => sum + parseFloat(entry[3] || 0), 0);

            document.getElementById('weekly-summary').textContent = `$${weeklyDeduction.toFixed(2)}`;

        }

        function refreshDashboard() {

            const revenue = calculateMonthlyRevenue();

            const expenses = calculateMonthlyExpenses();

            const miles = calculateMonthlyMiles();

            document.getElementById('total-revenue-month').textContent = `$${revenue.toFixed(2)}`;

            document.getElementById('total-expenses-month').textContent = `$${expenses.toFixed(2)}`;

            document.getElementById('net-profit-month').textContent = `$${(revenue - expenses).toFixed(2)}`;

            document.getElementById('total-miles-month').textContent = miles;

        }

        function getThisWeekData(data) {

            const now = new Date();

            const weekStart = new Date(now.setDate(now.getDate() - now.getDay()));

            return data.filter(entry => {

                const entryDate = new Date(entry[0]);

                return entryDate >= weekStart;

            });

        }

        function calculateMonthlyRevenue() {

            const rover = JSON.parse(localStorage.getItem('michaels_dogs_rover') || '[]');

            const venmo = JSON.parse(localStorage.getItem('michaels_dogs_venmo') || '[]');

            const cash = JSON.parse(localStorage.getItem('michaels_dogs_cash') || '[]');

            return getThisMonthData(rover).reduce((sum, entry) => sum + parseFloat(entry[1] || 0), 0) +

                   getThisMonthData(venmo).reduce((sum, entry) => sum + parseFloat(entry[2] || 0), 0) +

                   getThisMonthData(cash).reduce((sum, entry) => sum + parseFloat(entry[2] || 0), 0);

        }

        function calculateMonthlyExpenses() {

            const expenses = JSON.parse(localStorage.getItem('michaels_dogs_expenses') || '[]');

            const mileage = JSON.parse(localStorage.getItem('michaels_dogs_mileage') || '[]');

            return getThisMonthData(expenses).reduce((sum, entry) => sum + parseFloat(entry[2] || 0), 0) +

                   getThisMonthData(mileage).reduce((sum, entry) => sum + parseFloat(entry[3] || 0), 0);

        }

        function calculateMonthlyMiles() {

            const mileage = JSON.parse(localStorage.getItem('michaels_dogs_mileage') || '[]');

            const thisMonth = getThisMonthData(mileage);

            return thisMonth.reduce((sum, entry) => sum + parseFloat(entry[1] || 0), 0).toFixed(1);

        }

        function getThisMonthData(data) {

            const now = new Date();

            const currentMonth = now.getMonth();

            const currentYear = now.getFullYear();

            return data.filter(entry => {

                const entryDate = new Date(entry[0]);

                return entryDate.getMonth() === currentMonth && entryDate.getFullYear() === currentYear;

            });

        }
