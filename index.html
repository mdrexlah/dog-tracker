    function saveToStorage(key, value) {

            showStatus('setup-status', 'Storage warning: Data may not persist.', 'warning');

            return false;

        }

    function deleteEntry(type, entryId) {

    function updateRecentEntries() {

        const types = ['mileage', 'rover', 'venmo', 'cash', 'expenses'];

        

        types.forEach(type => {

            const container = document.getElementById(`recent-${type}`);

            if (container) {

                const entries = getFromStorage(type) || [];

                const recent = entries.slice(-5).reverse();

                

                if (recent.length === 0) {

                    container.innerHTML = 'No entries yet';

                    return;

                }

                

                container.innerHTML = recent.map(entry => {

                    const displayText = getEntryDisplayText(type, entry);

                    return `

                        <div class="entry-item">

                            <div class="entry-details">

                                <div class="entry-date">${entry.date}</div>

                                <div>${displayText}</div>

                                <div class="entry-amount">$${(entry.amount || entry.deduction || 0).toFixed(2)}</div>

                            </div>

                            <button onclick="deleteEntry('${type}', ${entry.id})" class="delete-btn">×</button>

                        </div>

                    `;

                }).join('');

            }

        });

    }


    function getEntryDisplayText(type, entry) {

            case 'rover':

                return `Rover deposit - ${entry.period || 'Payment'}`;

            case 'venmo':

                return `${entry.client} - ${entry.service}`;

            case 'cash':

            case 'expenses':

            default:

                return 'Entry';

        }

    }

    function getThisWeekData(data) {

        const now = new Date();

        const weekStart = new Date(now.setDate(now.getDate() - now.getDay()));

        return data.filter(entry => new Date(entry.date) >= weekStart);

    }


    function getThisMonthData(data) {

        const now = new Date();

        const monthStart = new Date(now.getFullYear(), now.getMonth(), 1);

        return data.filter(entry => new Date(entry.date) >= monthStart);

    }


    function getPeriodData(period) {

        const allData = {

            mileage: getFromStorage('mileage') || [],

            rover: getFromStorage('rover') || [],

            venmo: getFromStorage('venmo') || [],

            cash: getFromStorage('cash') || [],

            expenses: getFromStorage('expenses') || []

        };

        

        const now = new Date();

        let startDate;

        

        switch(period) {

            case 'current':

                startDate = new Date(now.getFullYear(), now.getMonth(), 1);

                break;

            case 'last':

                startDate = new Date(now.getFullYear(), now.getMonth() - 1, 1);

                const endDate = new Date(now.getFullYear(), now.getMonth(), 0);

                Object.keys(allData).forEach(key => {

                    allData[key] = allData[key].filter(entry => {

                        const entryDate = new Date(entry.date);

                        return entryDate >= startDate && entryDate <= endDate;

                    });

                });

                return allData;

            case 'ytd':

                startDate = new Date(now.getFullYear(), 0, 1);

                break;

            default:

                startDate = new Date(now.getFullYear(), now.getMonth(), 1);

        }

        

        Object.keys(allData).forEach(key => {

            allData[key] = allData[key].filter(entry => new Date(entry.date) >= startDate);

        });

        

        return allData;

    }


    function getPeriodLabel(period) {

        const now = new Date();

        switch(period) {

            case 'current':

                return now.toLocaleDateString('en-US', { month: 'long', year: 'numeric' });

            case 'last':

                const lastMonth = new Date(now.getFullYear(), now.getMonth() - 1);

                return lastMonth.toLocaleDateString('en-US', { month: 'long', year: 'numeric' });

            case 'ytd':

                return `Year to Date ${now.getFullYear()}`;

            default:

                return 'Current Period';

        }

    }
